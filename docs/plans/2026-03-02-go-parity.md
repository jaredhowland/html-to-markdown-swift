### Task 10: Smart Escaping System

This is the largest change. Replace the unused `escapeMarkdown()` with a real two-pass smart escaping system.

**Reference:** Go library uses placeholder bytes in text nodes, then a post-render pass decides per-placeholder whether to insert `\` based on context.

**How it works:**
1. Text transform: For each potential markdown special character, insert a placeholder Unicode char (`\u{FFF9}`) before it
2. Post-render unescaping: Walk the final string and for each placeholder+char pair, decide whether to insert `\` based on context rules

**Context rules (when to actually escape):**
- `#` at start of a line (preceded only by spaces/newlines) followed by 1-6 `#`s and then space → escape (ATX header)
- `=`/`-` with non-empty line above it and all same char → escape (setext header)
- `-`/`*`/`+` at start of line (preceded only by spaces/newlines) followed by space → escape (unordered list)
- digit+`.` at start of line → escape (ordered list)
- `*`/`_` not followed by whitespace → escape (emphasis)
- `[` when balanced with `]` followed by `(` or `[` → escape (link)

**Step 3: Create `Sources/SmartEscaping.swift`**

```swift
import Foundation

// Placeholder character used to mark potential escape candidates during text processing.
// Using Unicode Private Use Area character that won't appear in normal content.
let escapePlaceholder: Character = "\u{FFF9}"
let escapePlaceholderString = String(escapePlaceholder)

/// Characters that may need escaping in Markdown context
let markdownSpecialChars: Set<Character> = [
    "\\", "`", "*", "_", "{", "}", "[", "]", "(", ")", "#", "+", "-", ".", "!", "~"
]

/// Mark potential escape candidates in text with placeholder
func markEscapeCandidates(_ text: String) -> String {
    var result = ""
    for char in text {
        if markdownSpecialChars.contains(char) {
            result.append(escapePlaceholder)
        }
        result.append(char)
    }
    return result
}

/// Process the final markdown output, deciding which placeholders become `\` escapes
func applySmartEscaping(_ markdown: String) -> String {
    let chars = Array(markdown)
    var changes = [Int: Bool]() // index → should escape
    
    var i = 0
    while i < chars.count {
        guard chars[i] == escapePlaceholder else {
            i += 1
            continue
        }
        guard i + 1 < chars.count else {
            i += 1
            continue
        }
        let nextChar = chars[i + 1]
        let shouldEscape = checkShouldEscape(chars: chars, placeholderIndex: i, char: nextChar)
        changes[i] = shouldEscape
        i += 1
    }
    
    var result = ""
    var i2 = 0
    while i2 < chars.count {
        if chars[i2] == escapePlaceholder {
            if changes[i2] == true {
                result.append("\\")
            }
            // Skip the placeholder itself
            i2 += 1
            continue
        }
        result.append(chars[i2])
        i2 += 1
    }
    return result
}

/// Determine whether the placeholder at `placeholderIndex` (followed by `char`) should emit `\`
private func checkShouldEscape(chars: [Character], placeholderIndex: Int, char: Character) -> Bool {
    switch char {
    case "#":
        return isAtxHeader(chars: chars, index: placeholderIndex + 1)
    case "=", "-":
        return isSetextHeader(chars: chars, index: placeholderIndex + 1)
    case "-", "*", "+":
        return isUnorderedList(chars: chars, index: placeholderIndex + 1)
    case ".":
        return isOrderedList(chars: chars, index: placeholderIndex + 1)
    case "*", "_":
        return isEmphasis(chars: chars, index: placeholderIndex + 1)
    case "[":
        return isBracketLink(chars: chars, index: placeholderIndex + 1)
    default:
        return false
    }
}

// MARK: - Context Checkers

/// Returns true if `#` at index is at start of line and would form an ATX heading
private func isAtxHeader(chars: [Character], index: Int) -> Bool {
    guard chars[index] == "#" else { return false }
    
    // Walk backward to check only spaces/newlines before this #
    for i in stride(from: index - 2, through: 0, by: -1) { // -2 because -1 is the placeholder
        if chars[i] == "\n" { break }
        if chars[i] == " " || chars[i] == escapePlaceholder { continue }
        return false // Non-space, non-newline before the # → not at start of line
    }
    
    // Count consecutive # signs
    var hashCount = 0
    var j = index
    while j < chars.count && (chars[j] == "#" || chars[j] == escapePlaceholder) {
        if chars[j] == "#" { hashCount += 1 }
        if hashCount > 6 { return false }
        j += 1
    }
    
    // Must be followed by space, tab, newline, or end
    if j >= chars.count { return true }
    return chars[j] == " " || chars[j] == "\t" || chars[j] == "\n" || chars[j] == "\r"
}

/// Returns true if `=` or `-` at index would form a setext heading underline
private func isSetextHeader(chars: [Character], index: Int) -> Bool {
    let ch = chars[index]
    guard ch == "=" || ch == "-" else { return false }
    
    var newlineCount = 0
    for i in stride(from: index - 2, through: 0, by: -1) {
        if chars[i] == escapePlaceholder || chars[i] == " " { continue }
        if chars[i] == "\n" {
            newlineCount += 1
            continue
        }
        if newlineCount == 0 { return false } // Same line, not a setext underline
        if newlineCount == 1 { return true }  // Line above has content → setext heading
        return false
    }
    return false
}

/// Returns true if `-`, `*`, or `+` at index is at start of line followed by space
private func isUnorderedList(chars: [Character], index: Int) -> Bool {
    let ch = chars[index]
    guard ch == "-" || ch == "*" || ch == "+" else { return false }
    
    // Check nothing non-space before this on the current line
    for i in stride(from: index - 2, through: 0, by: -1) {
        if chars[i] == "\n" { break }
        if chars[i] == " " || chars[i] == escapePlaceholder { continue }
        return false
    }
    
    // Must be followed by space or end
    let next = index + 1
    if next >= chars.count { return true }
    return chars[next] == " " || chars[next] == "\t"
}

/// Returns true if `.` at index is part of `N.` at start of line
private func isOrderedList(chars: [Character], index: Int) -> Bool {
    guard chars[index] == "." else { return false }
    
    // Char before (skipping placeholder) must be a digit
    var prevIdx = index - 2 // -2 because -1 is the placeholder
    while prevIdx >= 0 && chars[prevIdx] == escapePlaceholder { prevIdx -= 1 }
    guard prevIdx >= 0, chars[prevIdx].isNumber else { return false }
    
    // Walk back: only digits and spaces allowed (and newline as stop)
    for i in stride(from: prevIdx, through: 0, by: -1) {
        if chars[i] == "\n" { break }
        if chars[i] == " " || chars[i] == escapePlaceholder { continue }
        if chars[i].isNumber { continue }
        return false
    }
    
    // Must be followed by space or end
    let next = index + 1
    if next >= chars.count { return true }
    return chars[next] == " " || chars[next] == "\t"
}

/// Returns true if `*` or `_` would form emphasis (not followed by whitespace)
private func isEmphasis(chars: [Character], index: Int) -> Bool {
    let ch = chars[index]
    guard ch == "*" || ch == "_" else { return false }
    let next = index + 1
    if next >= chars.count { return false }
    let nextChar = chars[next]
    return nextChar != " " && nextChar != "\t" && nextChar != "\n" && nextChar != "\r"
}

/// Returns true if `[` would form a link (followed eventually by `](`)
private func isBracketLink(chars: [Character], index: Int) -> Bool {
    guard chars[index] == "[" else { return false }
    // Look for matching ] followed by (
    var depth = 1
    var i = index + 1
    while i < chars.count && depth > 0 {
        if chars[i] == "[" { depth += 1 }
        if chars[i] == "]" {
            depth -= 1
            if depth == 0 {
                let next = i + 1
                if next < chars.count && (chars[next] == "(" || chars[next] == "[") {
                    return true
                }
            }
        }
        i += 1
    }
    return false
}
```

```swift
import Foundation
import SwiftSoup

/// Collapse whitespace in the DOM before rendering, following HTML whitespace rules:
/// - In inline context: collapse consecutive whitespace chars to single space
/// - At block boundaries: remove whitespace-only text nodes
func collapseWhitespaceInDOM(_ document: Document, isBlock: (Element) -> Bool) throws {
    try collapseNode(document, isBlock: isBlock, insideBlock: true)
}

private func collapseNode(_ node: Node, isBlock: (Element) -> Bool, insideBlock: Bool) throws {
    if let textNode = node as? TextNode {
        let text = textNode.getWholeText()
        if insideBlock {
            // Inside a block: normalize whitespace to collapsed form
            let collapsed = collapseWhitespaceString(text)
            if collapsed.isEmpty || collapsed == " " {
                // Check if it's surrounded by block elements — if so, remove it
                let isOnlyWhitespace = text.allSatisfy { $0.isWhitespace }
                if isOnlyWhitespace {
                    let prevIsBlock = isBlockSibling(textNode.previousSibling(), isBlock: isBlock)
                    let nextIsBlock = isBlockSibling(textNode.nextSibling(), isBlock: isBlock)
                    if prevIsBlock || nextIsBlock {
                        try textNode.remove()
                        return
                    }
                }
            }
            try textNode.text(collapsed)
        }
        return
    }
    
    guard let element = node as? Element else { return }
    let nodeIsBlock = isBlock(element)
    
    for child in element.getChildNodes() {
        try collapseNode(child, isBlock: isBlock, insideBlock: nodeIsBlock || insideBlock)
    }
}

private func isBlockSibling(_ node: Node?, isBlock: (Element) -> Bool) -> Bool {
    guard let node = node else { return false }
    if let element = node as? Element { return isBlock(element) }
    return false
}

private func collapseWhitespaceString(_ text: String) -> String {
    // Replace runs of whitespace (\t, \n, \r, space) with a single space
    var result = ""
    var lastWasSpace = false
    for char in text {
        if char.isWhitespace {
            if !lastWasSpace {
                result.append(" ")
                lastWasSpace = true
            }
        } else {
            result.append(char)
            lastWasSpace = false
        }
    }
    return result
}
```
