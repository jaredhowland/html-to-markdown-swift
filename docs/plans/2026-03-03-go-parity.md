func isAtStartOfLine(chars: [Character], idx: Int) -> Bool {

/// Mirrors Go's converter/prioritized.go
public let PriorityEarly: Int = 100
public let PriorityStandard: Int = 500
public let PriorityLate: Int = 1000

struct PrioritizedValue<V> {
    let value: V
    let priority: Int
}

typealias PrioritizedSlice<V> = [PrioritizedValue<V>]

func prioritized<V>(_ v: V, _ priority: Int) -> PrioritizedValue<V> {
    return PrioritizedValue(value: v, priority: priority)
}

extension Array {
    func sortedByPriority<V>() -> [PrioritizedValue<V>] where Element == PrioritizedValue<V> {
        return self.sorted { $0.priority < $1.priority }
    }
}

public enum RenderStatus {
    case tryNext   // RenderTryNext — handler did not handle this node

public typealias HandleTextTransformFunc = (Context, String) -> String

    /// Called once to register all handlers. Mirrors Go's Init(conv *Converter) error.
    func initialize(conv: Converter) throws

// Sources/converter/ctx.swift
import Foundation
import SwiftSoup

/// Context mirrors Go's converter.Context interface.
/// Carries per-conversion state and provides rendering/escaping helpers.
public class Context {
    let conv: Converter
    var domain: String
    private var values: [String: Any] = [:]
    var globalState: [String: Any] = [:]

    init(conv: Converter, domain: String) {
        self.conv = conv
        self.domain = domain
    }

    private init(copying other: Context) {
        self.conv = other.conv
        self.domain = other.domain
        self.values = other.values
        self.globalState = other.globalState
    }

    // MARK: - URL assembly

    /// Mirrors Go's ctx.AssembleAbsoluteURL.
    public func assembleAbsoluteURL(tagName: String, rawURL: String) -> String {
        return defaultAssembleAbsoluteURL(rawURL: rawURL, domain: domain)
    }

    // MARK: - Tag type

    /// Mirrors Go's ctx.GetTagType.
    public func getTagType(_ tagName: String) -> (TagType, Bool) {
        let t = conv.registry.getType(for: tagName)
        // Return false if tagName is completely unknown (not registered and not in fallback lists)
        return (t, true)
    }

    // MARK: - Rendering

    /// Mirrors Go's ctx.RenderNodes.
    public func renderNodes(_ w: StringWriter, _ nodes: [Node]) {
        conv.handleRenderNodes(ctx: self, w: w, nodes: nodes)
    }

    /// Mirrors Go's ctx.RenderChildNodes.
    public func renderChildNodes(_ w: StringWriter, _ n: Node) {
        let children = n.getChildNodes()
        conv.handleRenderNodes(ctx: self, w: w, nodes: children)
    }

    // MARK: - Escaping

    /// Mirrors Go's ctx.EscapeContent.
    public func escapeContent(_ content: String) -> String {
        return conv.escapeContent(content)
    }

    /// Mirrors Go's ctx.UnEscapeContent.
    public func unEscapeContent(_ content: String) -> String {
        return conv.unEscapeContent(content)
    }

    // MARK: - Key-value store (mirrors Go's context.WithValue / ctx.Value)

    /// Mirrors Go's ctx.WithValue — returns a new Context with the added value.
    public func withValue(_ key: String, _ val: Any) -> Context {
        let copy = Context(copying: self)
        copy.values[key] = val
        return copy
    }

    public func getValue<V>(_ key: String) -> V? {
        return values[key] as? V
    }

    // MARK: - Global state (mirrors Go's SetState/GetState/UpdateState)

    public func getState<V>(_ key: String) -> V? {
        return globalState[key] as? V
    }
    public func setState<V>(_ key: String, val: V) {
        globalState[key] = val
    }
    public func updateState<V>(_ key: String, fn: (V?) -> V) {
        let current = globalState[key] as? V
        globalState[key] = fn(current)
    }
}

// Sources/converter/converter.swift
import Foundation
import SwiftSoup

public class Converter {
    let lock = NSLock()
    var err: Error? = nil
    var registeredPlugins: [String] = []

    var preRenderHandlers: PrioritizedSlice<HandlePreRenderFunc> = []
    var renderHandlers:    PrioritizedSlice<HandleRenderFunc>    = []
    var postRenderHandlers:PrioritizedSlice<HandlePostRenderFunc> = []
    var textTransformHandlers: PrioritizedSlice<HandleTextTransformFunc> = []

    var markdownChars: Set<Character> = []
    var unEscapeHandlers: PrioritizedSlice<HandleUnEscapeFunc> = []

    var tagTypesMap: [String: PrioritizedSlice<TagType>] = [:]

    var escapeMode: EscapeMode = .smart

    // Per-conversion options (domain, selectors) set via ConverterOption
    var domain: String = ""
    var excludeSelectors: [String] = []
    var includeSelector: String? = nil

    /// registry is still used for getType(for:) fallback — backed by tagTypesMap
    var registry: TagTypeRegistry { return TagTypeRegistry(backingMap: tagTypesMap) }

    /// Mirrors Go's conv.Register field.
    public var Register: Register { return Register(conv: self) }

    public init(options: [ConverterOption] = []) {
        processOptions(options)
    }

    /// Convenience init matching current Swift API — accepts plugins directly.
    public convenience init(plugins: [Plugin], options: [ConverterOption] = []) {
        self.init(options: options)
        for plugin in plugins {
            do {
                try Register.plugin(plugin)
            } catch {
                setError(error)
            }
        }
    }

    func setError(_ e: Error) {
        lock.lock(); defer { lock.unlock() }
        err = e
    }
    func getError() -> Error? {
        lock.lock(); defer { lock.unlock() }
        return err
    }

    private func processOptions(_ options: [ConverterOption]) {
        for option in options {
            switch option {
            case .domain(let url):
                domain = url
            case .excludeSelectors(let selectors):
                excludeSelectors = selectors
            case .includeSelector(let selector):
                includeSelector = selector
            case .escapeMode(let mode):
                escapeMode = mode
            case .tagTypeConfiguration(let config):
                var reg = TagTypeRegistry(backingMap: tagTypesMap)
                config(&reg)
                tagTypesMap = reg.types.mapValues { entry in
                    [PrioritizedValue(value: entry.type, priority: 500)]
                }
            case .customRenderers(let renderers):
                for (tag, renderer) in renderers {
                    Register.rendererFor(tag, .inline, { ctx, w, n in
                        if let result = try? renderer(n, self) {
                            w.writeString(result ?? "")
                            return .success
                        }
                        return .tryNext
                    }, priority: PriorityEarly)
                }
            case .plugins(let plugins):
                for plugin in plugins {
                    do { try Register.plugin(plugin) } catch { setError(error) }
                }
            }
        }
    }
}

    case disabled  // EscapeModeSmart    = "smart"

// Sources/converter/register.swift
import Foundation
import SwiftSoup

/// Register mirrors Go's converter.register struct.
/// All methods mutate the parent Converter's handler slices.
public struct Register {
    unowned let conv: Converter

    // MARK: - Plugin

    /// Mirrors Go's register.Plugin — calls plugin.initialize(conv) after validation.
    public func plugin(_ p: Plugin) throws {
        let pluginName = p.name
        guard !pluginName.isEmpty else {
            throw ConversionError.pluginError("the plugin has no name")
        }
        conv.lock.lock()
        conv.registeredPlugins.append(pluginName)
        conv.lock.unlock()

        try p.initialize(conv: conv)
    }

    // MARK: - PreRenderer

    /// Mirrors Go's register.PreRenderer.
    public func preRenderer(_ fn: @escaping HandlePreRenderFunc, priority: Int) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        conv.preRenderHandlers.append(prioritized(fn, priority))
    }

    // MARK: - Renderer

    /// Mirrors Go's register.Renderer.
    public func renderer(_ fn: @escaping HandleRenderFunc, priority: Int) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        conv.renderHandlers.append(prioritized(fn, priority))
    }

    /// Mirrors Go's register.RendererFor — registers a TagType AND a Renderer filtered to tagName.
    public func rendererFor(_ tagName: String, _ tagType: TagType, _ fn: @escaping HandleRenderFunc, priority: Int) {
        self.tagType(tagName, tagType, priority: priority)
        self.renderer({ ctx, w, n in
            guard (n as? Element)?.tagName().lowercased() == tagName.lowercased()
                || n.nodeName().lowercased() == tagName.lowercased()
            else { return .tryNext }
            return fn(ctx, w, n)
        }, priority: priority)
    }

    // MARK: - PostRenderer

    /// Mirrors Go's register.PostRenderer.
    public func postRenderer(_ fn: @escaping HandlePostRenderFunc, priority: Int) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        conv.postRenderHandlers.append(prioritized(fn, priority))
    }

    // MARK: - TextTransformer

    /// Mirrors Go's register.TextTransformer.
    public func textTransformer(_ fn: @escaping HandleTextTransformFunc, priority: Int) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        conv.textTransformHandlers.append(prioritized(fn, priority))
    }

    // MARK: - Escaping

    /// Mirrors Go's register.EscapedChar.
    public func escapedChar(_ chars: Character...) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        for c in chars { conv.markdownChars.insert(c) }
    }

    /// Mirrors Go's register.UnEscaper.
    public func unEscaper(_ fn: @escaping HandleUnEscapeFunc, priority: Int) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        conv.unEscapeHandlers.append(prioritized(fn, priority))
    }

    // MARK: - Tag type

    /// Mirrors Go's register.TagType.
    public func tagType(_ tagName: String, _ type: TagType, priority: Int) {
        conv.lock.lock(); defer { conv.lock.unlock() }
        let tag = tagName.lowercased()
        conv.tagTypesMap[tag, default: []].append(prioritized(type, priority))
    }
}

        lock.lock(); defer { lock.unlock() }

// Sources/converter/render.swift
import Foundation
import SwiftSoup

extension Converter {

    /// Mirrors Go's handleRenderNodes — renders a list of nodes to a writer.
    func handleRenderNodes(ctx: Context, w: StringWriter, nodes: [Node]) {
        for node in nodes {
            handleRenderNode(ctx: ctx, w: w, node: node)
        }
    }

    /// Mirrors Go's handleRenderNode — dispatches to text handler, render handlers, or fallback.
    @discardableResult
    func handleRenderNode(ctx: Context, w: StringWriter, node: Node) -> RenderStatus {
        let name = node.nodeName().lowercased()

        // A: #text nodes
        if name == "#text", let textNode = node as? TextNode {
            return handleRenderText(ctx: ctx, w: w, node: textNode)
        }

        // B: registered render handlers (sorted by priority)
        for handler in getRenderHandlers() {
            let status = handler(ctx, w, node)
            if status == .success { return .success }
        }

        // C: fallback
        return handleRenderFallback(ctx: ctx, w: w, node: node)
    }

    private func handleRenderFallback(ctx: Context, w: StringWriter, node: Node) -> RenderStatus {
        let tagName = node.nodeName().lowercased()
        let tagType = getTagType(tagName)

        if tagType == .block {
            w.writeString("\n\n")
        }
        ctx.renderChildNodes(w, node)
        if tagType == .block {
            w.writeString("\n\n")
        }
        return .success
    }

    private func handleRenderText(ctx: Context, w: StringWriter, node: TextNode) -> RenderStatus {
        var content = node.getWholeText()

        conv.Register.renderer({ ctx, w, n in

/// Remove all nodes whose tag type is .remove.
/// Mirrors Go's base.preRenderRemove.
private func removeTaggedNodes(_ doc: Document, ctx: Context) {
    func remove(_ node: Node) {
        var toRemove: [Node] = []
        for child in node.getChildNodes() {
            let name = child.nodeName().lowercased()
            if ctx.conv.getTagType(name) == .remove {
                toRemove.append(child)
            } else {
                remove(child)
            }
        }
        for n in toRemove { try? n.remove() }
    }
    remove(doc)
}

/// Fallback renderer: renders block elements with surrounding newlines, inline elements pass through.
/// Registered at PriorityLate+100 so plugin renderers (PriorityStandard) run first.
private func renderFallback(ctx: Context, w: StringWriter, node: Node, conv: Converter) -> RenderStatus {
    let tagName = node.nodeName().lowercased()
    let tagType = conv.getTagType(tagName)

    if tagType == .remove { return .success }  // already removed in pre-render

    if tagType == .block { w.writeString("\n\n") }
    ctx.renderChildNodes(w, node)
    if tagType == .block { w.writeString("\n\n") }
    return .success
}

extension Converter {
    func escapeContent(_ content: String) -> String {
        if escapeMode == .disabled { return content }
        var result = ""
        for char in content {
            if char == "\u{0000}" {
                result.append("\u{FFFD}")
                continue
            }
            if checkIsEscapedChar(char) {
                result.append(escapePlaceholder)
            }
            result.append(char)
        }
        return result
    }

    func unEscapeContent(_ content: String) -> String {
        if escapeMode == .disabled { return content }
        let chars = Array(content)
        let handlers = getUnEscapeHandlers()

        // Two-pass: determine which placeholders to escape
        var actions = Array(repeating: false, count: chars.count)
        var i = 0
        while i < chars.count {
            if chars[i] != escapePlaceholder { i += 1; continue }
            if i + 1 >= chars.count { break }
            for handler in handlers {
                let skip = handler(chars, i + 1)
                if skip != -1 {
                    actions[i] = true
                    i += skip
                    break
                }
            }
            i += 1
        }

        var result = ""
        for (idx, char) in chars.enumerated() {
            if char == escapePlaceholder {
                if actions[idx] { result.append("\\") }
                // always skip the placeholder itself
            } else {
                result.append(char)
            }
        }
        return result
    }
}

        right = String(right.dropLast())

    var leftExtra = ""
    var remaining = right
    while let first = remaining.first, first.isWhitespace {

        remaining = String(remaining.dropFirst())
